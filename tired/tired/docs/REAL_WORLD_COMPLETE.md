### 2. 智能搜索過濾
```swift
// 關鍵詞匹配
let queryLower = trimmedQuery.lowercased()
return SearchResults(
    posts: posts.filter { 
        $0.summary.lowercased().contains(queryLower) || 
        $0.content.lowercased().contains(queryLower) 
    },
    // ...其他類型
)
```

### 3. 數據緩存策略
```swift
// 檢查緩存
if let cached = conversationsStore[userId] {
    return cached.sorted { $0.lastMessageTimestamp > $1.lastMessageTimestamp }
}
// 載入 Mock 數據並緩存
let conversations = await MockDataProvider.shared.mockConversations(for: userId)
conversationsStore[userId] = conversations
return conversations
```
